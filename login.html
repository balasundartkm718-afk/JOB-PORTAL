const express = require('express'); const router = express.Router(); const db = require('../db'); router.post('/add', (req, res) => { const { title, description, company, location, posted_by } = req.body; const sql = "INSERT INTO jobs (title, description, company, location, posted_by) VALUES (?, ?, ?, ?, ?)"; db.query(sql, [title, description, company, location, posted_by], (err, result) => { if (err) return res.send(err); res.send("Job Added"); }); }); router.get('/', (req, res) => { db.query("SELECT * FROM jobs", (err, result) => { if (err) return res.send(err); res.json(result); }); }); module.exports = router;
